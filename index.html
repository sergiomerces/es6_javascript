Currrying

Uma única função que recebe um parâmetro continua retornando uma nova função, salvando umdos valores na memória para reutilização.

function soma(a, b)
{
return a + b;
}

console.log(soma(2, 3));
console.log(soma(2, 4));
console.log(soma(2, 5));
console.log(soma(2, 6));

function soma(a)
{
return function(b)
{
return a + b;
}
}

const soma2 = soma(2);

console.log(soma2(3));
console.log(soma2(4));
console.log(soma2(5));
console.log(soma2(6));
